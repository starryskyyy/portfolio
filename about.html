public void binarySearch(int key, int[] ary, int arrayLength) {    int lo, hi, mid;    lo = 0;    hi = arrayLength - 1;
   while (lo <= hi) {      mid = (lo + hi) / 2; // find a midpoint
     if (ary[mid] == key) //if we key was on the midpoint        return mid;      // if key greater than midpoint set a new midpoint      if (ary[mid] < key)        hi = mid - 1;      // if key less than midpoint set a new midpoint      else        lo = mid + 1;    }    return -1; }
